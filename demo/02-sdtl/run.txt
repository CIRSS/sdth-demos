=========================== bash cell | import sdtl ============================

# Destroy the dataset
geist destroy --dataset kb --quiet

# Import the SDTL JSON-LD file
geist create --dataset kb --inputformat json-ld --inputfile data/sdtl.jsonld --infer owl
